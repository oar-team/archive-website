# Bind configuration for OAR virtual cluster
###

# Bind config sample
view "subnet-1" {
	match-clients { 10.1.1.0/24; };
	zone "oar.vpn" {
		type master;
		file "db.oar.vpn.subnet-1";
	}
}
view "subnet-2" {
	match-clients { 10.1.2.0/24; };
	zone "oar.vpn" {
		type master;
		file "db.oar.vpn.subnet-2";
	}
}
view "subnet-3" {
	match-clients { 10.1.3.0/24; };
	zone "oar.vpn" {
		type master;
		file "db.oar.vpn.subnet-3";
	}
}
